class VideoGames:

    """

    @staticmethod
    def japanplatform():
        """
        This static method will calculate the market share
        of each of the four chosen platforms for Japan


        """

        # get the csv file and read it
        vg_sales_reader = pd.read_csv("vgsales.csv", engine="python")
        # make a dataframe
        vg_df = pd.DataFrame(vg_sales_reader)
        # get the rows
        jps3_rows = vg_sales_reader[vg_sales_reader.Platform == "PS3"]  # get ps3 rows
        jwii_rows = vg_sales_reader[vg_sales_reader.Platform == "Wii"]  # get wii rows
        jx360_rows = vg_sales_reader[vg_sales_reader.Platform == "X360"]  # get xbox rows
        jps2_rows = vg_sales_reader[vg_sales_reader.Platform == "PS2"]  # get ps 2 rows
        # get the market shares for each platform
        jps3_global_sales_avg = (jps3_rows["JP_Sales"].sum() / len(
            jps3_rows["JP_Sales"])) * 1000000  # get the market share number
        jwii_global_sales_avg = (jwii_rows["JP_Sales"].sum() / len(jwii_rows["JP_Sales"])) * 1000000
        jx360_global_sales_avg = (jx360_rows["JP_Sales"].sum() / len(jx360_rows["JP_Sales"])) * 1000000
        jps2_global_sales_avg = (jps2_rows["JP_Sales"].sum() / len(jps2_rows["JP_Sales"])) * 1000000
        # print the dollar amounts
        print("\n\nJapan ps2 sales: $"+str(int(jps2_global_sales_avg)) +
              "00 million")  # print the sales for each platform in millions
        print("Japan ps3 sales: $"+ str(int(jps3_global_sales_avg))+ "00 million")  # print sales for ps 3
        print("Japan wii sales: $"+ str(int(jwii_global_sales_avg))+ "00 million")  # print sales for wii
        print("Japan 360 sales: $"+str(int(jx360_global_sales_avg))+ "00 hundred thousand")  # print sales for xbox360

        # make a list of all the information for the pie
        jpglobal_sales_list = [jps3_global_sales_avg, jx360_global_sales_avg, jwii_global_sales_avg,
                               jps2_global_sales_avg, ]

        labels = "PS3", "XBox360", "Wii", "PS2"  # label the plot
        explode = (0.0, 0.1, 0.0, 0.1)
        colors = ['#FF99CC', '#E0E0E0', '#66B2FF', '#ffcc99', ]  # color the plot
        # make the pie chart
        j_sales_pie = plt.pie(jpglobal_sales_list, explode=explode, labels=labels, colors=colors, autopct='%1.0f%%')

        plt.title(" Japan Sales Average by Platform \n (in millions)")  # title the pie chart
        plt.show()

    @staticmethod

        """ This static method will visualize the genres
        and their global sales in the form of a bar chart
        """
        # make a set
        vg_sales_reader = pd.read_csv("vgsales.csv", engine="python")
        # make a dataframe
        vg_df = pd.DataFrame(vg_sales_reader)
        genre_set = set()
        genre_names = []
        # for loop to add to a set
        for line in vg_sales_reader["Genre"]:
            genre_set.add(line)  # append, eliminate repeats

        genre_names = ['Action', 'Adventure', 'Fighting', 'Misc', 'Platform', 'Puzzle', 'Racing',
                       'Role-Playing', 'Shooter', 'Simulation', 'Sports', 'Strategy']  # name the genres
        # group by genre
        by_genre = pd.DataFrame(vg_sales_reader.groupby("Genre").sum())  # group by genre
        by_genre_global_sales = by_genre["Global_Sales"]  # genre and global sales
        genre_names = ['Action', 'Adventure', 'Fighting', 'Misc', 'Platform', 'Puzzle', 'Racing',
                       'Role-Playing', 'Shooter', 'Simulation', 'Sports', 'Strategy']  # genre names list
        # make a plot and add colors
        genresales = plt.bar(genre_names, by_genre_global_sales,
                             color=["#7aff73", '#73fff1', '#ff9473', '#ffda73', '#b273ff', '#ff73c0'])
        # title the graph
        plt.title("Global Sale Totals by Genre \n (in millions)")
        # colors = ['#FF99CC','#E0E0E0','#66B2FF','#ffcc99',]

        plt.xticks(rotation=50)
        plt.show()  # appear neat

def make_gb_plot_graph():
    """
    This function creates a graph of global sales from a range of years
    using a publisher input.
    
    """
    publisher_input = input("Select a publisher/company: ")

    publisher_only = vg_sales_reader[vg_sales_reader.Publisher == publisher_input]
    publisher_sorted = publisher_only.sort_values(['Year'])

    year_only = []
    gb_sales_only = []
    for year in publisher_sorted['Year']:
        year_only.append(year)

    for gb_sale in publisher_sorted['Global_Sales']:
        gb_sales_only.append(gb_sale)

    plt.figure(figsize=(10, 6))
    plt.plot(year_only, gb_sales_only)
    plt.xlabel("Year")
    plt.ylabel("Sales in Millions")
    plt.title("Global Sales of Company from 1980 to 2015")
    plt.show()
